import dask.array as da

arrays = []
for array in npy_files:
    file = os.path.join(DIR, array)
    data = np.load(file, mmap_mode="r+")
    arr = da.from_array(data, chunks=(5_000_000, 8))
    arrays.append(arr)
del data, arr

a = da.concatenate(arrays)
